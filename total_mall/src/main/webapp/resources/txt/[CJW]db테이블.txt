CREATE TABLE tb_product (
    p_idx    int    unsigned not null auto_increment,
    m_idx    int    unsigned not null,
    category    varchar(30)    NOT NULL,
    product_name    varchar(100)    NOT NULL,
    product_code varchar(50),
    brand    varchar(30)    NULL,
    summary    varchar(50)    NOT NULL,
    detail    varchar(1000)    NOT NULL,
    start_date    date    not NULL,
    end_date    date    not NULL,
    origin_file_name1    varchar(50),
    save_file_name1    varchar(50),
    origin_file_name2    varchar(50),
    save_file_name2    varchar(50),
    price    int unsigned NOT NULL,
    p_status decimal(1) default 0,
    discount_rate    int unsigned,
    quantity    int    unsigned,
    constraint tb_product_pk primary key(p_idx),
    p_reg_date timestamp default now(),
    p_last_modified_date timestamp default now(),
    read_cnt int unsigned
);
